- [ ] **Step 1: Write failing tests for the content contract**

Create this complete initial test file. It uses temporary fixture directories and the exported `validateContent(root, options)` function:

```javascript
import assert from 'node:assert/strict';
import {afterEach, test} from 'node:test';
import {mkdtemp, mkdir, rm, writeFile} from 'node:fs/promises';
import {join} from 'node:path';
import {tmpdir} from 'node:os';
import {validateContent} from '../scripts/validate-content.mjs';

const roots = [];
const launchSlugs = [
  '/cases/microsoft-multi-agent-reference-architecture',
  '/cases/openai-agents-sdk',
  '/cases/langgraph-supervisor',
  '/cases/google-adk-a2a',
  '/cases/aws-cli-agent-orchestrator',
];

afterEach(async () => {
  await Promise.all(roots.splice(0).map((root) => rm(root, {recursive: true, force: true})));
});

async function fixtureRoot() {
  const root = await mkdtemp(join(tmpdir(), 'agentic-atlas-'));
  roots.push(root);
  return root;
}

async function writeDocument(root, name, body) {
  const directory = join(root, 'cases');
  await mkdir(directory, {recursive: true});
  await writeFile(join(directory, `${name}.mdx`), body, 'utf8');
}

function validCase(slug, difficulty = 'intermediate') {
  const title = slug.split('/').at(-1);
  return `---
title: ${title}
slug: ${slug}
content_type: case
status: reviewed
difficulty: ${difficulty}
analyzed_at: 2026-07-20
source_cutoff: 2026-07-20
confidence: high
domains:
  - agent-runtime
agent_patterns:
  - supervisor
protocols: []
quality_attributes:
  - recoverability
tags:
  - 多智能体
official_sources:
  - https://example.com/official
---

# ${title}
`;
}

test('rejects a case with missing required front matter', async () => {
  const root = await fixtureRoot();
  await writeDocument(root, 'broken', `---\ntitle: Broken\ncontent_type: case\n---\n`);

  const {errors} = await validateContent(root);

  assert.ok(errors.some((error) => error.includes('broken.mdx') && error.includes('slug')));
});

test('rejects unknown enum values', async () => {
  const root = await fixtureRoot();
  await writeDocument(root, 'invalid-enum', validCase('/cases/invalid-enum', 'expert-plus'));

  const {errors} = await validateContent(root);

  assert.ok(errors.some((error) => error.includes('difficulty') && error.includes('expert-plus')));
});

test('reports missing required launch cases', async () => {
  const root = await fixtureRoot();
  await writeDocument(root, 'openai-agents-sdk', validCase(launchSlugs[1]));

  const {errors} = await validateContent(root, {requireLaunchCases: true});

  assert.ok(errors.some((error) => error.includes(launchSlugs[0])));
  assert.equal(errors.filter((error) => error.includes('Missing launch case')).length, 4);
});

test('accepts five valid launch cases with official sources', async () => {
  const root = await fixtureRoot();
  await Promise.all(launchSlugs.map((slug) => writeDocument(root, slug.split('/').at(-1), validCase(slug))));

  const {errors} = await validateContent(root, {requireLaunchCases: true});

  assert.deepEqual(errors, []);
});
```

```yaml
name: Deploy to GitHub Pages

on:
```
